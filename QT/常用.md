1、获取当前时间
```cpp
// 打印当前时间
qDebug() << "Current time:" << QDateTime::currentDateTime().toString("yyyy-MM-dd hh:mm:ss.zzz");

// 或者分开日期和时间
qDebug() << "Current date:" << QDateTime::currentDateTime().date().toString("yyyy-MM-dd");
qDebug() << "Current time:" << QDateTime::currentDateTime().time().toString("hh:mm:ss.zzz");

qDebug() << "Current time:" << QTime::currentTime().toString("hh:mm:ss.zzz");

// 格式化输出
qDebug("%s", QTime::currentTime().toString("hh:mm:ss.zzz").toLocal8Bit().constData());
```

void MainWindow::bubbleSort(int *srcArr, int size)
{
    bool swapped;
    for(int i = 0;i < size - 1;i++)
    {
        swapped = false;
        for(int j = 0;j<size -1 -i;j++ )
        {
            if (srcArr[j] > srcArr[j + 1])
            {
                // 交换相邻元素
                int temp = srcArr[j];
                srcArr[j] = srcArr[j + 1];
                srcArr[j + 1] = temp;
                swapped = true;
            }
        }
        // 如果这一轮没有发生交换，说明数组已经有序，提前结束
        if (!swapped)
            break;
    }
}

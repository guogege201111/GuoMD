```cpp
void MainWindow::bubbleSort(int *srcArr, int size)
{
    bool swapped;
    for(int i = 0;i < size - 1;i++)
    {
        swapped = false;
        for(int j = 0;j<size -1 -i;j++ )
        {
            if (srcArr[j] > srcArr[j + 1])
            {
                // 交换相邻元素
                int temp = srcArr[j];
                srcArr[j] = srcArr[j + 1];
                srcArr[j + 1] = temp;
                swapped = true;
            }
        }
        // 如果这一轮没有发生交换，说明数组已经有序，提前结束
        if (!swapped)
            break;
    }
}
```

```cpp
void MainWindow::selectSort(int *srcArr, int size)
{
    for(int i = 0; i < size - 1; i++)
    {
        int minIndex = i;
        // 找到未排序部分中最小值
        for(int j = i + 1; j < size; j++)
        {
            if(srcArr[j] < srcArr[minIndex])
               minIndex = j;
        }
        // 如果未排序最小值下标和已排序最小值不一样，则交换两者位置
        if(minIndex != i)
        {
            int temp;
            temp = srcArr[minIndex];
            srcArr[minIndex] = srcArr[i];
            srcArr[i] = temp;
        }
    }
}
```

3、插入
```cpp
void MainWindow::insertSort(int *srcArr, int size)
{
    // 相当于第一个数不动，从第二个数开始计算
    for(int i = 1;i<size;i++)
    {
        // 暂存待插入的数
        int key = srcArr[i];
        // 待插入数的前一个的下标，即已排序数的最后一个
        int j = i-1;
        // 已排序数从后往前开始，如果大于即将插入的数，则将已排序的数的最后一个后移一个
        // 遇到第一个 <= key（或到头）就停止，把 key 放到它后面的位置（即 j+1）。
        while(srcArr[j] > key && j>0)
        {
            srcArr[j+1] = srcArr[j];
            j--;
        }
        srcArr[j+1] = key;
    }
}
```

```CPP
// 希尔排序

void MainWindow::shellSort(int *srcArr, int size) 
{
    int n = size;
    // 初始增量 gap = n/2，每次减半，直到 gap=1
    for (int gap = n / 2; gap > 0; gap /= 2) 
    {
        // 对每个子序列进行插入排序
        for (int i = gap; i < n; i++) 
        {
            // 待插入元素
            int temp = srcArr[i];      
            int j = i;
            // 在子序列中向前寻找插入位置
            while (j >= gap && srcArr[j - gap] > temp) 
            {
                srcArr[j] = srcArr[j - gap]; // 向后移动元素
                j -= gap;
            }
            srcArr[j] = temp; // 插入
        }
    }
}
```

1、将任意数据结构保存到本地bin文件
```cpp
#include <iostream>
#include <fstream>

QString binPath = QApplication::applicationDirPath() + QString("/")+QString("bin/") + QString("%1_%2.bin").arg(QDateTime::currentDateTime().toString("HH-mm-ss-zzz"),QString::number(transferred));

std::ofstream ofs;

ofs.open(binPath.toStdString(), std::ios::out | std::ios::binary);

// 写入buffer中的数据到文件
ofs.write(reinterpret_cast<const char*>(mUsbPackage), sizeof(sUsbPack));
//关闭文件
ofs.close();

```
